---
title: "List Edges"
api: "GET /v1/boards/{boardId}/edges"
description: "Returns a paginated list of edges on a board."
---

<ParamField path="boardId" type="string" required>
  Board UUID.
</ParamField>

<ParamField query="limit" type="integer" default={50}>
  Number of items per page (1-100).
</ParamField>

<ParamField query="cursor" type="string">
  Opaque cursor from a previous response.
</ParamField>

<ParamField query="order" type="string" default="desc">
  Sort direction by `created_at`.
</ParamField>

<ParamField query="source_node_id" type="string">
  Filter edges by source node UUID.
</ParamField>

<ParamField query="target_node_id" type="string">
  Filter edges by target node UUID.
</ParamField>

<RequestExample>
